使用Docker部署Trilium

docker-compose

version: "3.7"

services:
  trilium:
    image: zadam/trilium
    restart: unless-stopped
    ports:
      - "8080:8080"
    volumes:
      - /data/trilium/data:/home/node/trilium-data

docker-compose up -d
